// the setup function runs once when you press reset or power the board
void setup() {
    pinMode(2, OUTPUT);  
    pinMode(3, OUTPUT);
    pinMode(4, OUTPUT);
    pinMode(5, OUTPUT);
}

// the loop function runs over and over again forever
void loop() {
  
  digitalWrite(2, 1); //LSB
  digitalWrite(3, 0); 
  digitalWrite(4, 1); 
  digitalWrite(5, 0); //MSB
}
